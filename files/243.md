# Shortest Word Distance [E]
Given an array of strings wordsDict and two different strings that already exist in the array word1 and word2, return the shortest distance between these two words in the list.


Example 1:
Input: wordsDict = ["practice", "makes", "perfect", "coding", "makes"], word1 = "coding", word2 = "practice"
Output: 3

Example 2:
Input: wordsDict = ["practice", "makes", "perfect", "coding", "makes"], word1 = "makes", word2 = "coding"
Output: 1

```python
class Solution:
    def shortestDistance(self, wordsDict: List[str], word1: str, word2: str) -> int:
        prev = -1
        res = len(wordsDict)
        for i, w in enumerate(wordsDict):
            if w == word1 or w == word2:
                if prev >= 0 and w != wordsDict[prev]:
                    # only possible to update if current word is different than prev occurence
                    res = min(res, i-prev)
                prev = i
        return res
```
or
```python
class Solution:
    def shortestDistance(self, wordsDict: List[str], word1: str, word2: str) -> int:
        w1_loc, w2_loc = -1, -1
        res = len(wordsDict)
        for i, w in enumerate(wordsDict):
            if w == word1:
                w1_loc = i
            elif w == word2:
                w2_loc = i
            if w1_loc > -1 and w2_loc > -1:
                res = min(res, abs(w1_loc-w2_loc))
        return res
```
