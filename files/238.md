### Solution
```python
class Solution(object):
    def productExceptSelf(self, nums):
        """
        :type nums: List[int]
        :rtype: List[int]
        """
        # caveat, if init res with 0, then res[0] will be 0 (when doing right->left population we do *=)
        res = [1] * len(nums)
        leftMul = 1
        for i in range(1, len(nums)):
            leftMul *= nums[i - 1]
            res[i] = leftMul

        rightMul = 1
        for j in reversed(range(len(nums) - 1)):
            rightMul *= nums[j + 1]
            res[j] *= rightMul

        return res
```
