# Pacific Atlantic Water Flow

### Solution DFS
Create a set of co-ordinates from where we can reach pacific ocean. lets call it P_Set.
Create a set of co-ordinates from where we can reach atlantic ocean. lets call it A_Set
Our answer would be intersection of A_Set and P_Set
Calculate P_Set:

Add all the boundary co-ordiates to the set along the pacific ocean
Perform BFS to add to the set if curr height > parent height.
```python
class Solution:
    def pacificAtlantic(self, heights: List[List[int]]) -> List[List[int]]:
        m, n = len(heights), len(heights[0])
        dirs = [(-1, 0), (1, 0), (0, -1), (0, 1)]
        pSet, aSet = set(), set()
        for j in range(n):
            pSet.add((0, j))
            aSet.add((m-1, j))
        for i in range(m):
            pSet.add((i, 0))
            aSet.add((i, n-1))
        
        def dfs(i, j, s):
            for di, dj in dirs:
                ni, nj = i + di, j + dj
                if 0 <= ni < m and 0 <= nj < n and (ni, nj) not in s and heights[ni][nj] >=  heights[i][j]:
                    s.add((ni, nj))
                    dfs(ni, nj, s)
        
        for j in range(n):
            dfs(0, j, pSet)
            dfs(m-1, j, aSet)
        for i in range(m):
            dfs(i, 0, pSet)
            dfs(i, n-1, aSet)
        res = []
        for i, j in pSet:
            if (i, j) in aSet:
                res.append((i, j))
        return res
```

BFS
```python
from collections import deque
from typing import List

class Solution:
    def pacificAtlantic(self, heights: List[List[int]]) -> List[List[int]]:
        if not heights or not heights[0]: return []
        m, n = len(heights), len(heights[0])
        dirs = [(-1, 0), (1, 0), (0, -1), (0, 1)]
        
        # 1. 初始化 Set 和 Queue
        pSet, aSet = set(), set()
        pQueue, aQueue = deque(), deque()

        # 2. 边界多源初始化
        for i in range(m):
            # 左边缘（太平洋）
            pSet.add((i, 0))
            pQueue.append((i, 0))
            # 右边缘（大西洋）
            aSet.add((i, n-1))
            aQueue.append((i, n-1))
            
        for j in range(n):
            # 上边缘（太平洋）
            if (0, j) not in pSet:
                pSet.add((0, j))
                pQueue.append((0, j))
            # 下边缘（大西洋）
            if (m-1, j) not in aSet:
                aSet.add((m-1, j))
                aQueue.append((m-1, j))

        # 3. 定义统一的扩散逻辑
        def bfs(queue, visited):
            while queue:
                r, c = queue.popleft()
                for dr, dc in dirs:
                    nr, nc = r + dr, c + dc
                    # 关键修复点 1：检查 nr, nc 是否在 visited 中（你之前在 aQueue 循环里查了 pSet）
                    # 关键修复点 2：高度必须是 >=，即“逆流而上”
                    if (0 <= nr < m and 0 <= nc < n and 
                        (nr, nc) not in visited and 
                        heights[nr][nc] >= heights[r][c]):
                        visited.add((nr, nc))
                        queue.append((nr, nc))

        # 4. 分别执行两次 BFS
        bfs(pQueue, pSet)
        bfs(aQueue, aSet)
        
        # 5. 返回交集
        return [list(pos) for pos in pSet if pos in aSet]
```
