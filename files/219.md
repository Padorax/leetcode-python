### Solution
```python
class Solution:
    def containsNearbyDuplicate(self, nums: List[int], k: int) -> bool:
        numPos = dict()
        for i, num in enumerate(nums):
            if num in numPos and -k <= numPos[num] - i <=k:
                return True
            else:
                numPos[num] = i # update index
        
        return False
```
