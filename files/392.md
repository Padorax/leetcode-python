```python
class Solution:
    def isSubsequence(self, s: str, t: str) -> bool:
        if len(s) > len(t): return False
        if len(s) == 0: return True
        
        subSeqPos = 0
        for c in t:
            if c == s[subSeqPos]:
                subSeqPos += 1
                if subSeqPos == len(s):
                    return True
        
        return False
```
