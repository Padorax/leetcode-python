# Network Delay Time
You are given a network of n nodes, labeled from 1 to n. You are also given times, a list of travel times as directed edges times[i] = (ui, vi, wi), where ui is the source node, vi is the target node, and wi is the time it takes for a signal to travel from source to target.

We will send a signal from a given node k. Return the minimum time it takes for all the n nodes to receive the signal. If it is impossible for all the n nodes to receive the signal, return -1.

### Solution 1 DFS
```python
class Solution:
    def networkDelayTime(self, times: List[List[int]], n: int, k: int) -> int:
        neighbors = dict()
        for u, v, t in times:
            if u not in neighbors:
                neighbors[u] = []
            neighbors[u].append((v, t))
        
        mp = {}
        def dfs(time, node):
            mp[node] = time
            for nbr, t in neighbors.get(node, []):
                # ! allow visit `visited` node ONLY if it's a shorter path (effectively prevents backwards)
                if nbr not in mp or mp[nbr] > time+t:
                    dfs(time+t, nbr)
        dfs(0, k)
        return -1 if len(mp) < n else max(mp.values())
```

### Dijkstra
```python
class Solution:
    def networkDelayTime(self, times: List[List[int]], n: int, k: int) -> int:
        edges = collections.defaultdict(list)
        for u, v, w in times:
            edges[u].append((v, w))

        minHeap = [(0, k)]
        visit = set()
        t = 0
        while minHeap:
            w1, n1 = heapq.heappop(minHeap)
            if n1 in visit:
                continue
            visit.add(n1)
            t = w1

            for n2, w2 in edges[n1]:
                if n2 not in visit:
                    heapq.heappush(minHeap, (w1 + w2, n2))
        return t if len(visit) == n else -1
```
time: O(E*log(V)), space: O(V+E)
V: number of vertices, E: number of edges
 
or: instead of simply using visit, use a time_mp to record weight so far
```python
import heapq

class Solution:
    def networkDelayTime(self, times: list[list[int]], n: int, k: int) -> int:
        # 1. 建图
        adj = [[] for _ in range(n)]
        for u, v, w in times:
            adj[u-1].append((v-1, w))
        
        # 2. Dijkstra 初始化
        # time_mp 记录每个节点收到的最短信号时间
        time_mp = {}
        pq = [(0, k-1)] # (时间, 节点)
        
        while pq:
            t, u = heapq.heappop(pq)
            
            # 如果该节点已经处理过且当前时间更长，跳过
            if u in time_mp: #!!! no need for cur_time >= time_mp[node]
                continue
            
            # 第一次弹出时，一定是该节点的最短路径
            time_mp[u] = t
            
            for v, weight in adj[u]:
                if v not in time_mp:
                    heapq.heappush(pq, (t + weight, v))
        
        # 3. 结果判断
        # 只有当收到的节点数等于 n 时，返回所有最短路径中的最大值
        if len(time_mp) == n:
            return max(time_mp.values())
        return -1
```

optimized
```python
import heapq
class Solution:
    def networkDelayTime(self, times: List[List[int]], n: int, k: int) -> int:
        adj = [[] for _ in range(n)]
        for prev, nxt, t in times:
            adj[prev-1].append((t, nxt-1))
        
        time_mp = {k-1: 0}
        h = [(0, k-1)]
        res = 0
        while h:
            cur_time, node = heapq.heappop(h)
            if node in time_mp and cur_time > time_mp[node]:
                continue
            
            res = cur_time

            for t, nbr in adj[node]:
                if nbr not in time_mp or t + cur_time < time_mp[nbr]:
                    time_mp[nbr] = cur_time+t
                    heapq.heappush(h, (t+cur_time, nbr))
        
        return res if len(time_mp) == n else -1
```
### Bellman-Ford
```python
class Solution:
    def networkDelayTime(self, times: List[List[int]], n: int, k: int) -> int:
        dist = [float('inf')] * n
        dist[k - 1] = 0
        for _ in range(n - 1):
            for u, v, w in times:
                if dist[u - 1] + w < dist[v - 1]:
                    dist[v - 1] = dist[u - 1] + w
        max_dist = max(dist)
        return max_dist if max_dist < float('inf') else -1
```
time: O(E*V), space: O(V)

### SPFA (shortest path faster algorithm)
```python
class Solution:
    def networkDelayTime(self, times, n, k):
        adj = defaultdict(list)
        for u, v, w in times:
            adj[u].append((v, w))

        dist = {node: float("inf") for node in range(1, n + 1)}
        q = deque([(k, 0)])
        dist[k] = 0

        while q:
            node, time = q.popleft()
            if dist[node] < time:
                continue
            for nei, w in adj[node]:
                if time + w < dist[nei]:
                    dist[nei] = time + w
                    q.append((nei, time + w))

        res = max(dist.values())
        return res if res < float('inf') else -1
```
Time complexity: O(V+E) in average case, O(V∗E) in worst case.
Space complexity: O(V+E)
