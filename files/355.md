# Design Twitter
Implement a simplified version of Twitter which allows users to post tweets, follow/unfollow each other, and view the 10 most recent tweets within their own news feed.

Users and tweets are uniquely identified by their IDs (integers).

Implement the following methods:
* `Twitter()` Initializes the twitter object.
* `void postTweet(int userId, int tweetId)` Publish a new tweet with ID tweetId by the user userId. You may assume that each tweetId is unique.
* `List<Integer> getNewsFeed(int userId)` Fetches at most the 10 most recent tweet IDs in the user's news feed. Each item must be posted by users who the user is following or by the user themself. Tweets IDs should be ordered from most recent to least recent.
* `void follow(int followerId, int followeeId)` The user with ID `followerId` follows the user with ID `followeeId`.
* `void unfollow(int followerId, int followeeId)` The user with ID `followerId` unfollows the user with ID `followeeId`.

### Solution - Heap
similar to merge K sorted list, We use a min-heap (priority queue) because:
* We push only the latest tweet of each followee.
* Each tweet has a timestamp (count), where smaller means more recent.
* We repeatedly extract the most recent tweet and then push the next tweet from that same user.
* This is similar to merging K sorted lists efficiently.
```python
class Twitter:

    def __init__(self):
        # 使用全局计数器来模拟时间戳。数值越小，代表时间越近（最新）。
        self.count = 0
        # 存储推文：userId -> [[time1, tweetId1], [time2, tweetId2], ...]
        self.tweetMap = defaultdict(list)  
        # 存储关注关系：userId -> {followeeId1, followeeId2, ...}
        self.followMap = defaultdict(set)  

    def postTweet(self, userId: int, tweetId: int) -> None:
        # 1. 记录推文及其时间戳
        self.tweetMap[userId].append([self.count, tweetId])
        
        # 2. 优化：每个用户只需要保留自己最新的 10 条推文
        # 因为 getNewsFeed 只要总共前 10 条，一个人出再多也不可能超过 10 条被选中
        if len(self.tweetMap[userId]) > 10:
            self.tweetMap[userId].pop(0) # 删掉最早的推文
            
        # 3. 时间戳递减，确保最新的推文 count 最小
        self.count -= 1

    def getNewsFeed(self, userId: int) -> List[int]:
        res = []
        minHeap = [] # 这里实际作为“大顶堆”效果使用，因为 count 是负的
        
        # 0. 潜规则：自己必须关注自己，才能看到自己的推文
        self.followMap[userId].add(userId)
        
        # 1. 将所有关注的人的最新的那一条推文放入堆中
        # 这一步是“多路归并”的初始化
        for followeeId in self.followMap[userId]:
            if followeeId in self.tweetMap:
                # 拿到该用户最后一条（最新的）推文的索引
                index = len(self.tweetMap[followeeId]) - 1
                count, tweetId = self.tweetMap[followeeId][index]
                # 堆存入：[时间戳, 推文ID, 发送者ID, 下一条推文的索引]
                heapq.heappush(minHeap, [count, tweetId, followeeId, index - 1])

        # 2. 经典的 K 路归并过程：取堆顶（最新），然后补位
        while minHeap and len(res) < 10:
            count, tweetId, followeeId, index = heapq.heappop(minHeap)
            res.append(tweetId) # 记录结果
            
            # 如果这个发推者还有更旧的推文，把下一条补进堆里继续竞争
            if index >= 0:
                count, tweetId = self.tweetMap[followeeId][index]
                heapq.heappush(minHeap, [count, tweetId, followeeId, index - 1])

        return res

    def follow(self, followerId: int, followeeId: int) -> None:
        self.followMap[followerId].add(followeeId)

    def unfollow(self, followerId: int, followeeId: int) -> None:
        # 注意：不能取关自己（虽然面试题通常不考这点，但加个判断更稳）
        if followeeId in self.followMap[followerId] and followerId != followeeId:
            self.followMap[followerId].remove(followeeId)
```

Optimization
在 getNewsFeed 里加入了一个两级堆（Heap）优化。

很多人的写法是直接把所有关注者的推文丢进一个堆里。但你的代码多想了一层：如果关注的人非常多（比如关注了 1000 个人），那么堆的大小就是 1000。你的代码通过 if len(self.followMap[userId]) >= 10 试图先筛选出“最新推文最靠前的 10 个人”，从而控制最终参与归并的堆的大小。
```python
class Twitter:
    def __init__(self):
        # 全局计时器，由于 Python heapq 是小顶堆，用负数递减可以实现“数值越小，时间越新”
        self.count = 0
        self.tweetMap = defaultdict(list)
        self.followMap = defaultdict(set)

    def postTweet(self, userId: int, tweetId: int) -> None:
        # 插入新推文
        self.tweetMap[userId].append([self.count, tweetId])
        # 【内存优化】：因为 NewsFeed 只取前 10 条，单人存超过 10 条在逻辑上是冗余的
        if len(self.tweetMap[userId]) > 10:
            self.tweetMap[userId].pop(0) # 弹出最旧的
        self.count -= 1

    def getNewsFeed(self, userId: int) -> List[int]:
        res = []
        minHeap = []
        # 默认关注自己，否则看不到自己的推文
        self.followMap[userId].add(userId)

        # --- 优化逻辑开始 ---
        # 如果关注的人很多（>=10），先筛选出这群人里推文“最新”的前 10 个人
        if len(self.followMap[userId]) >= 10:
            maxHeap = [] # 这里其实是想选出 time 最大的前 10 个
            for followeeId in self.followMap[userId]:
                if followeeId in self.tweetMap:
                    index = len(self.tweetMap[followeeId]) - 1
                    count, tweetId = self.tweetMap[followeeId][index]
                    # 这里的技巧：用 [-count] 配合 minHeap 模拟选出最新的人
                    heapq.heappush(maxHeap, [-count, tweetId, followeeId, index - 1])
                    if len(maxHeap) > 10:
                        heapq.heappop(maxHeap) # 弹出较旧的人
            
            # 将筛选出的 10 个人的最新推文转入真正的合并堆
            while maxHeap:
                count, tweetId, followeeId, index = heapq.heappop(maxHeap)
                heapq.heappush(minHeap, [-count, tweetId, followeeId, index])
        else:
            # 如果关注的人少，直接全部入堆参与归并
            for followeeId in self.followMap[userId]:
                if followeeId in self.tweetMap:
                    index = len(self.tweetMap[followeeId]) - 1
                    count, tweetId = self.tweetMap[followeeId][index]
                    heapq.heappush(minHeap, [count, tweetId, followeeId, index - 1])
        # --- 优化逻辑结束 ---

        # 【多路归并】：K-Way Merge 核心步骤
        # 每次从堆中取出全局最新的推文，然后把该用户的下一条推文补进来
        while minHeap and len(res) < 10:
            count, tweetId, followeeId, index = heapq.heappop(minHeap)
            res.append(tweetId)
            if index >= 0: # 如果该用户还有更旧的推文，继续入堆
                count, tweetId = self.tweetMap[followeeId][index]
                heapq.heappush(minHeap, [count, tweetId, followeeId, index - 1])

        return res

    def follow(self, followerId: int, followeeId: int) -> None:
        self.followMap[followerId].add(followeeId)

    def unfollow(self, followerId: int, followeeId: int) -> None:
        if followeeId in self.followMap[followerId]:
            # 这里的细节：不建议取关自己，否则 getNewsFeed 逻辑会乱
            if followerId != followeeId:
                self.followMap[followerId].remove(followeeId)

```
