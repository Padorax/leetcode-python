### Solution
适合用栈操作运算：遇到数字则入栈；遇到算符则取出栈顶两个数字进行计算，并将结果压入栈中
```python
import operator


class Solution:
    def evalRPN(self, tokens: List[str]) -> int:
        opt = {
            "+": operator.add,
            "-": operator.sub,
            "*": operator.mul,
            "/": operator.truediv,
        }
        s = []
        for token in tokens:
            if token in opt:
                s.append(int(opt[token](s.pop(-2), s.pop(-1))))
            else:
                s.append(int(token))
        return s[0]
```
or
```python
class Solution:
    def evalRPN(self, tokens: List[str]) -> int:
        opt = {
            "+": (lambda x, y: x + y),
            "-": (lambda x, y: x - y),
            "*": (lambda x, y: x * y),
            "/": (lambda x, y: (x // y
            if x*y >= 0 or x % y == 0 else x // y + 1))
        }
        stack = []
        for token in tokens:
            if token not in opt:
                stack.append(int(token))
            else:
                #note: cannot do stack.pop(-1), stack.pop(-2)
                # after pop(-1) previous -2 is now -1
                stack.append(opt[token] (stack.pop(-2), stack.pop(-1)))
        
        return stack[0]
```
