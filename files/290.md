### Word Pattern

### Solution
```python
class Solution:
    def wordPattern(self, pattern: str, s: str) -> bool:
        wordToPattern = dict()
        used_pattern = set()
        words = s.split()
    
        if len(words) != len(pattern):
            return False
            
        for i, word in enumerate(words):
            if word not in wordToPattern:
                if pattern[i] in used_pattern:
                    return False
                wordToPattern[word] = pattern[i]
                used_pattern.add(pattern[i])
            elif wordToPattern[word] != pattern[i]:
                return False
        
        return True
```

```python
def wordPattern(pattern: str, s: str) -> bool:
    ws = s.split()
    if len(pattern) != len(ws):
        return False
    d1 = {}
    d2 = {}
    for a, b in zip(pattern, ws):
        if (a in d1 and d1[a] != b) or (b in d2 and d2[b] != a):
            return False
        d1[a] = b
        d2[b] = a
    return True
```
