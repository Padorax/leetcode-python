### Solution Backtracing
```python
class Solution(object):
    def permute(self, nums):
        """
        :type nums: List[int]
        :rtype: List[List[int]]
        """
        res = []
        def dfs(path, res, count):
            if count == 0:
                res.append(path)
                return
            for num in nums:
                if num not in path:
                    dfs(path + [num], res, count - 1)
        def dfs2(path, res):
            if len(path) == len(nums):
                res.append(path.copy())
                return
            for num in nums:
                if num not in path:
                    path.append(num)
                    dfs2(path, res)
                    path.pop()
        dfs([], res, len(nums))
        return res
```

class Solution:
    def permute(self, nums: List[int]) -> List[List[int]]:
        res = []
        n = len(nums)
        def dfs(cur):
            if len(cur) == n:
                res.append(cur)
                return
            for num in nums:
                if num not in cur:
                    dfs(cur+[num])
        dfs([])
        return res

class Solution:
    def permute(self, nums: List[int]) -> List[List[int]]:
        if not nums:
            return [[]]
        perms = self.permute(nums[1:])
        res = []
        for perm in perms:
            for i in range(len(nums)):
                res.append(perm[:i]+[nums[0]]+perm[i:])
        return res
