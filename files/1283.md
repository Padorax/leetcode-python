class Solution:
    def smallestDivisor(self, nums: List[int], threshold: int) -> int:
        def getDivSum(div):
            res = 0
            for n in nums:
                v = math.ceil(n / div)
                res += v
            return res
        
        l, r = 1, max(nums)
        while l < r:
            mid = (l + r) // 2
            if getDivSum(mid) > threshold:
                l = mid + 1
            else:
                r = mid
        return l
