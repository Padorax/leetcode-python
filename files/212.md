# Word Search 2
Given a 2-D grid of characters board and a list of strings words, return all words that are present in the grid.

For a word to be present it must be possible to form the word with a path in the board with horizontally or vertically neighboring cells. The same cell may not be used more than once in a word.

### Solution
```python
from typing import List

class TrieNode:
    def __init__(self):
        self.children = {}
        self.word = None  # 存储完整的单词，非单词结尾则为 None

class Solution:
    def findWords(self, board: List[List[str]], words: List[str]) -> List[str]:
        # 1. 构建字典树
        root = TrieNode()
        for w in words:
            # 手动实现 addWord 逻辑
            cur = root
            for c in w:
                if c not in cur.children:
                    cur.children[c] = TrieNode()
                cur = cur.children[c]
            cur.word = w

        ROWS, COLS = len(board), len(board[0])
        res = []

        # 2. 定义回溯函数
        def dfs(r, c, node):
            char = board[r][c]
            # 如果当前字符不在字典树的分支里，直接回溯
            if char not in node.children:
                return
            
            # 找到匹配的子节点
            child_node = node.children[char]
            
            # 如果该节点存有单词，说明找到了一个结果
            if child_node.word:
                res.append(child_node.word)
                child_node.word = None  # 【优化1】找到后置空，防止结果重复
            
            # 标记当前单元格已访问（利用原棋盘，节省 set 空间）
            board[r][c] = "#"
            
            # 探索四个方向
            for dr, dc in [(r+1, c), (r-1, c), (r, c+1), (r, c-1)]:
                if 0 <= dr < ROWS and 0 <= dc < COLS:
                    dfs(dr, dc, child_node)
            
            # 回溯：恢复棋盘字符
            board[r][c] = char
            
            # 【优化2】动态剪枝：如果该子节点已经没有后续分支了，删掉它
            # 这意味着这棵树的“这条路”已经彻底挖完了，下次 DFS 不必再走
            if not child_node.children:
                node.children.pop(char)

        # 3. 遍历棋盘每一个起点
        for r in range(ROWS):
            for c in range(COLS):
                dfs(r, c, root)

        return res
```
