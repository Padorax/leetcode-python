### Solution
```python
import heapq
class Solution(object):
    def findKthLargest(self, nums, k):
        """
        :type nums: List[int]
        :type k: int
        :rtype: int
        """
        h = []
        for i in range(k):
            heapq.heappush(h, nums[i])
        # or
        # h = nums[:k]
        # heapq.heapify(h)

        for i in range(k, len(nums)):
            if nums[i] > h[0]:
                heapq.heappop(h)
                heapq.heappush(h, nums[i]) # equiv to heapreplace

        return h[0]
    
    def findKthLargest2(self, nums: List[int], k: int) -> int:
        h = []
        for num in nums:
            if len(h) < k or h[0] < num:
                heapq.heappush(h, num)
            if len(h) > k:
                heapq.heappop(h)
        return h[0]
```
