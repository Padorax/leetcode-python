# Minimum Time Difference
Given a list of 24-hour clock time points in "HH:MM" format, return the minimum minutes difference between any two time-points in the list.
 

Example 1:
Input: timePoints = ["23:59","00:00"]
Output: 1

Example 2:
Input: timePoints = ["00:00","23:59","00:00"]
Output: 0

### Solution
Caveat: clock is a cycle, so need to check distance betweeen first and last time additionally
```python
class Solution:
    def findMinDifference(self, timePoints: List[str]) -> int:
        timePoints.sort()
        def getDiff(t1, t2):
            if t1 == t2: return 0
            m1 = convertMin(t1)
            m2 = convertMin(t2)
            res = m2 - m1
            # if m1 == 0:
            #     res = min(res, convertMin("24:00")-m2) 
            return res

        def convertMin(t):
            return int(t[:2]) * 60 + int(t[3:])
        
        res = float('inf')
        for i in range(1, len(timePoints)):
            res = min(res, getDiff(timePoints[i-1], timePoints[i]))
            if res == 0: return res
        res = min(res, 24 * 60 - getDiff(timePoints[0], timePoints[-1]))
        print(timePoints[0], timePoints[-1], getDiff(timePoints[0], timePoints[-1]))
        return res
```

### Solution Bucket Sort
```python
class Solution:
    def findMinDifference(self, timePoints: List[str]) -> int:
        # create buckets array for the times converted to minutes
        minutes = [False] * (24 * 60)
        for time in timePoints:
            h, m = map(int, time.split(":"))
            min_time = h * 60 + m
            if minutes[min_time]:
                return 0
            minutes[min_time] = True
        prevIndex = float("inf")
        firstIndex = float("inf")
        lastIndex = float("inf")
        ans = float("inf")

        # find differences between adjacent elements in sorted array
        for i in range(24 * 60):
            if minutes[i]:
                if prevIndex != float("inf"):
                    ans = min(ans, i - prevIndex)
                prevIndex = i
                if firstIndex == float("inf"):
                    firstIndex = i
                lastIndex = i

        return min(ans, 24 * 60 - lastIndex + firstIndex)
```
