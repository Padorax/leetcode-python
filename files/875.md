# Koko Eating Banas
Binary Search
```python
class Solution:
    def minEatingSpeed(self, piles: List[int], h: int) -> int:
        def check(k):
            #return sum((x + k - 1) // k for x in piles) <= h
            return sum((x - 1) // k + 1 for x in piles) <= h

        l, r = 1, max(piles)
        while l < r:
            mid = (l + r) // 2
            if not check(mid):
                l = mid + 1
            else:
                r = mid
        return l
```
