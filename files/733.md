class Solution:
    def floodFill(self, image: List[List[int]], sr: int, sc: int, color: int) -> List[List[int]]:
        def dfs(r, c, orig, color):
            image[r][c] = color
            if c > 0 and image[r][c-1] == orig:
                dfs(r, c-1, orig, color)
            if c < n-1 and image[r][c+1] == orig:
                dfs(r, c+1, orig, color)
            if r > 0 and image[r-1][c] == orig:
                dfs(r-1, c, orig, color)
            if r < m-1 and image[r+1][c] == orig:
                dfs(r+1, c, orig, color)
        
        if not image or not image[0]:
            return image
        m, n = len(image), len(image[0])
        if sr < 0 or sr >= m or sc < 0 or sc >= n:
            return image
        origColor = image[sr][sc]
        if origColor == color:
            return image
        dfs(sr, sc, origColor, color)
        return image
