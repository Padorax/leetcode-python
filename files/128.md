```python
class Solution(object):
    def longestConsecutive(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
#         map: k-num val-the length of lcs that ends or starts at num
        map = dict()
        ans = 0
        
        for num in nums:
            if num not in map:
                left = map[num-1] if num-1 in map else 0
                right = map[num+1] if num+1 in map else 0
                size = left + right + 1
                map[num] = size
                
                ans = ans if ans > size else size
                
                map[num-left] = size
                map[num+right] = size
            else:
            #   duplicate
                continue
        
        return ans
```

```python
class Solution:
    def longestConsecutive(self, nums: List[int]) -> int:
        seqCount = dict()
        res = 0
        for num in nums:
            if num in seqCount: 
                continue
            leftCnt = seqCount.get(num - 1, 0)
            rightCnt = seqCount.get(num + 1, 0)
            size = leftCnt + 1 + rightCnt
            seqCount[num-leftCnt] = size
            seqCount[num+rightCnt] = size
            seqCount[num] = size # <- do not miss this line, otherwise 1, 2, 4, 5, 3, 3, 2nd 3 does not know it exists already
            res = max(res, size)
        return res
```

```python
class Solution:
    def longestConsecutive(self, nums: List[int]) -> int:
        mp = defaultdict(int)
        res = 0

        for num in nums:
            if not mp[num]:
                mp[num] = mp[num - 1] + mp[num + 1] + 1
                mp[num - mp[num - 1]] = mp[num]
                mp[num + mp[num + 1]] = mp[num]
                res = max(res, mp[num])
        return res
```
