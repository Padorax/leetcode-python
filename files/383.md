### Ransom Note
```python
def canConstruct(ransomNote: str, magazine: str) -> bool:
    cnt = Counter(magazine)
    for c in ransomNote:
        cnt[c] -= 1
        if cnt[c] < 0:
            return False
    return True
```
or
```python
def canConstruct(ransomNote: str, magazine: str) -> bool:
    wordMap = dict()
    for w in magazine:
        wordMap[w] = wordMap.get(w, 0) + 1
    for w in ransomNote:
        if w in wordMap and wordMap[w] > 0:
            wordMap[w] -= 1
        else:
            return False
    
    return True
```
