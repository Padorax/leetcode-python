### Solution 1
DFS with no return
```python
class Solution:
    def sumNumbers(self, root: Optional[TreeNode]) -> int:
        # need to make res a instance variable instead of local var, in order to modify from inside dfs
        self.res = 0
        if not root: return 0
        def dfs(node, val):
            num = val * 10 + node.val
            if not node.left and not node.right:
                self.res += num

            if node.left:
                dfs(node.left, num)
            if node.right:
                dfs(node.right, num)
        dfs(root, 0)
        return self.res
```

note we *must handle leaf node explicitly*, below has issues:
```python
def sumNumbers(root: Optional[TreeNode]) -> int:

            self.res += val

        num = val * 10 + node.val

        dfs(node.left, num)
        dfs(node.right, num)
    dfs(root, 0)
    return self.res
```

### Solution 2 - dfs with return value
```python
class Solution:
    def sumNumbers(self, root: Optional[TreeNode]) -> int:
        return self.dfs(root, 0)
    def dfs(self, node, val):
        res = 0
        if not node.left and not node.right:
            return val * 10 + node.val
        s = val * 10 + node.val
        if node.left:
            res += self.dfs(node.left, s)
        if node.right:
            res += self.dfs(node.right, s)
        return res
```
